   CreateMap<THospital, MCIHospital>()
            .ForMember(dest => dest.Id, opt => opt.Ignore())
            .ForMember(dest => dest.HospitalId, opt => opt.MapFrom(src => src.Id))
            .ForMember(dest => dest.Name, opt => opt.MapFrom(src => src.Name ?? string.Empty))
            .ForMember(dest => dest.Abbreviation, opt => opt.MapFrom(src => src.CallName ?? string.Empty))
            .ForMember(dest => dest.Alias, opt => opt.MapFrom(src => src.Alias ?? string.Empty))
            .ForMember(dest => dest.Color, opt => opt.MapFrom(src => src.Color ?? string.Empty))
            .ForMember(dest => dest.DistanceSort, opt => opt.Ignore());

IEnumerable<THospital> tHospitals = _tHospitalRepository
                .GetByCondition(item => item.Color != null)
                .OrderBy(item => item.WGS84Geo?.Distance(new Point(referenceX, referenceY)) ?? double.MaxValue)
                .Take(9);

            List<MCIHospital> mciHospitals = new();

            int distanceSort = 1; // 初始化距離排序為 1

            foreach (var item in tHospitals)
            {
                MCIHospital mciHospital = _mapper.Map<MCIHospital>(item);
                mciHospital.MCICaseId = mciCase.Id;
                mciHospital.DistanceSort = distanceSort++; // 設定距離排序並增加索引計數器
                mciHospitals.Add(mciHospital);
            }
